expr      = num | string | symb | group | infix ;
symb      = name | list | symb, ( argList | parList ) ;
list      = '{',  seq, '}' ;
argList   = '[',  seq, ']' ;
parList   = '[[', seq, ']]';
seq       = nil | expr, seqTail ;
seqTail   = nil | ',', expr, seqTail ;
group     = '(' , expr, ')';
infix     = expr, oper, expr ;
number    = integer| float  ;
